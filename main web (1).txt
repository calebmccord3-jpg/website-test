  <script>
    document.getElementById('year').textContent = new Date().getFullYear();

    function saveEntry() {
      const text = document.getElementById('memoryText').value;
      if (!text.trim()) return;
      const entries = JSON.parse(localStorage.getItem('entries')) || [];
      entries.push({ id: Date.now(), text, date: new Date().toLocaleString() });
      localStorage.setItem('entries', JSON.stringify(entries));
      document.getElementById('memoryText').value = '';
      renderEntries();
    }

    function deleteEntry(id) {
      let entries = JSON.parse(localStorage.getItem('entries')) || [];
      entries = entries.filter(e => e.id !== id);
      localStorage.setItem('entries', JSON.stringify(entries));
      renderEntries();
    }

    function editEntry(id) {
      let entries = JSON.parse(localStorage.getItem('entries')) || [];
      const entry = entries.find(e => e.id === id);
      if (entry) {
        const newText = prompt('Edit your memory:', entry.text);
        if (newText !== null) {
          entry.text = newText;
          localStorage.setItem('entries', JSON.stringify(entries));
          renderEntries();
        }
      }
    }

    function renderEntries() {
      const entries = JSON.parse(localStorage.getItem('entries')) || [];
      const container = document.getElementById('entries');
      container.innerHTML = '';
      entries.forEach(e => {
        const div = document.createElement('div');
        div.className = 'post';
        div.innerHTML = `<p>${e.text}</p><small>${e.date}</small>`;

        const controls = document.createElement('div');
        controls.className = 'entry-controls';
        controls.innerHTML = `
          <button onclick="editEntry(${e.id})">Edit</button>
          <button onclick="deleteEntry(${e.id})">Delete</button>
        `;
        div.appendChild(controls);

        container.appendChild(div);
      });
    }

    renderEntries();

    function handleParent(inputId, captionId, previewId, storageKey) {
      const input = document.getElementById(inputId);
      const captionInput = document.getElementById(captionId);
      const preview = document.getElementById(previewId);

      input.addEventListener('change', () => {
        const file = input.files[0];
        if (file) {
          const reader = new FileReader();
          reader.onload = () => {
            localStorage.setItem(storageKey + '_photo', reader.result);
            renderParent(previewId, storageKey);
          };
          reader.readAsDataURL(file);
        }
      });

      captionInput.addEventListener('input', () => {
        localStorage.setItem(storageKey + '_caption', captionInput.value);
        renderParent(previewId, storageKey);
      });

      renderParent(previewId, storageKey);
    }

    function renderParent(previewId, storageKey) {
      const preview = document.getElementById(previewId);
      const dataUrl = localStorage.getItem(storageKey + '_photo');
      const caption = localStorage.getItem(storageKey + '_caption') || '';
      preview.innerHTML = '';
      if (dataUrl) {
        preview.innerHTML = `<img src="${dataUrl}" style="max-width:100%; border-radius:8px; margin-top:0.5rem;" />`;
      }
      if (caption) {
        const cap = document.createElement('div');
        cap.className = 'caption';
        cap.textContent = caption;
        preview.appendChild(cap);
      }
    }

    handleParent('chrisParents', 'chrisCaption', 'chrisPreview', 'chrisParents');
    handleParent('bradenParents', 'bradenCaption', 'bradenPreview', 'bradenParents');
    handleParent('jacobParents', 'jacobCaption', 'jacobPreview', 'jacobParents');
  </script>
